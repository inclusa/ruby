```ruby
=begin
	to_i converteix a sensers
	to_f converteix a flotant
	to_s converteix a cadenes
	
	Aquest mètodes es poden realitzar així, però són més estrictes:
	to_str
	to_int
=end

class HolaMundo
        def initialize()
        end
        def saluda()
        	nombre = "1.8"
        	nombre = nombre.to_f
        	valorDos = 2
        	valorDos = valorDos.to_s
        	puts nombre=valorDos
        end
end

objeto = HolaMundo.new()
objeto.saluda
gets()
```
